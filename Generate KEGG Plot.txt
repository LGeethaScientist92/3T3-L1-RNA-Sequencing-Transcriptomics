library(tidyverse)

# -----------------------------
# 1. Read combined KEGG table
# -----------------------------
kegg <- read_csv("PanelB_shared_pathways_used.csv", show_col_types = FALSE)

# -----------------------------
# 2. Clean / format
# -----------------------------
kegg <- kegg %>%
  mutate(
    Species = factor(Species, levels = c("Mouse", "Human")),
    neglog10_padj = -log10(p.adjust)
  )

# -----------------------------
# 3. Optional: keep only key pathways
#    Replace this list with your final chosen pathways
# -----------------------------
selected_pathways <- c(
  "PI3K-Akt signaling pathway",
  "AMPK signaling pathway",
  "PPAR signaling pathway",
  "Insulin signaling pathway",
  "Regulation of lipolysis in adipocytes",
  "FoxO signaling pathway",
  "HIF-1 signaling pathway",
  "AGE-RAGE signaling pathway in diabetic complications",
  "Focal adhesion",
  "Regulation of actin cytoskeleton",
  "Apelin signaling pathway",
  "mTOR signaling pathway",
  "Cholesterol metabolism",
  "Cell cycle"
)

kegg_plot <- kegg %>%
  filter(Pathway %in% selected_pathways)

# -----------------------------
# 4. Order pathways by max GeneRatio
# -----------------------------
pathway_order <- kegg_plot %>%
  group_by(Pathway) %>%
  summarise(max_ratio = max(GeneRatio, na.rm = TRUE), .groups = "drop") %>%
  arrange(max_ratio) %>%
  pull(Pathway)

kegg_plot <- kegg_plot %>%
  mutate(Pathway = factor(Pathway, levels = pathway_order))

# -----------------------------
# 5. Plot
# -----------------------------
p_kegg <- ggplot(kegg_plot, aes(x = GeneRatio, y = Pathway)) +
  geom_point(
    aes(size = Count, color = Species),
    alpha = 0.9
  ) +
  scale_color_manual(
    values = c("Mouse" = "#2C7FB8", "Human" = "#D7301F")
  ) +
  scale_size_continuous(range = c(3, 8)) +
  labs(
    title = "B  Cross-species KEGG pathway enrichment",
    x = "Gene ratio",
    y = NULL,
    color = NULL,
    size = "Gene count"
  ) +
  theme_classic(base_size = 11) +
  theme(
    plot.title = element_text(face = "bold", hjust = 0),
    axis.text.y = element_text(size = 9, color = "black"),
    axis.text.x = element_text(size = 9, color = "black"),
    axis.title.x = element_text(size = 10, color = "black"),
    legend.position = "right",
    legend.text = element_text(size = 9),
    panel.grid.major.x = element_line(color = "grey85", linewidth = 0.3),
    panel.grid.minor.x = element_blank(),
    panel.grid.major.y = element_blank()
  )

print(p_kegg)

# -----------------------------
# 6. Save
# -----------------------------
ggsave(
  "PanelB_KEGG_mouse_vs_human_clean.png",
  p_kegg,
  width = 8,
  height = 6.5,
  dpi = 300
)
