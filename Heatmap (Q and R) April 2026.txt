#!/usr/bin/env Rscript

suppressPackageStartupMessages({
  library(DESeq2)
  library(pheatmap)
  library(AnnotationDbi)
  library(org.Mm.eg.db)
  library(matrixStats)
})

# -------- Load --------
dds <- readRDS("/mnt/c/RNASeqProject/results/dds.rds")

# -------- Subset Q vs R --------
dds_rq <- dds[, dds$group %in% c("TreatedQ", "TreatedR")]
dds_rq$group <- droplevels(dds_rq$group)

# -------- Run DESeq2 --------
dds_rq <- DESeq(dds_rq)

# -------- DEGs: Q vs R --------
res <- results(dds_rq, contrast = c("group", "TreatedQ", "TreatedR"))
res <- res[!is.na(res$padj), ]

# -------- Clean Ensembl IDs --------
rownames(res) <- sub("\\..*", "", rownames(res))

# -------- Map to gene symbols --------
gene_symbols <- mapIds(
  org.Mm.eg.db,
  keys = rownames(res),
  column = "SYMBOL",
  keytype = "ENSEMBL",
  multiVals = "first"
)

res$symbol <- gene_symbols

# Remove NA + duplicates
res <- res[!is.na(res$symbol), ]
res <- res[order(res$padj), ]
res <- res[!duplicated(res$symbol), ]

# -------- Filter significant DEGs --------
res_sig <- res[res$padj < 0.05 & abs(res$log2FoldChange) > 1, ]

# -------- Select Top 30 --------
top30 <- head(res_sig[order(abs(res_sig$log2FoldChange), decreasing=TRUE), ], 30)
top_genes <- rownames(top30)

# -------- VST --------
vst_mat <- assay(vst(dds_rq, blind=TRUE))
rownames(vst_mat) <- sub("\\..*", "", rownames(vst_mat))

# -------- Subset matrix --------
mat <- vst_mat[top_genes, ]

# -------- Replace rownames with SYMBOL --------
rownames(mat) <- top30$symbol

# -------- Scale (Z-score) --------
mat <- mat - rowMeans(mat)

# -------- Annotation --------
sample_anno <- as.data.frame(colData(dds_rq)[, "group", drop=FALSE])

# -------- Heatmap --------
pheatmap(
  mat,
  annotation_col = sample_anno,
  fontsize_row = 7,
  fontsize_col = 9,
  main = "Top 30 DEGs: TreatedQ vs TreatedR",
  color = colorRampPalette(c("navy","white","firebrick3"))(50)
)

# -------- Save --------
png("/mnt/c/RNASeqProject/results/heatmap_top30_DEGs_QR.png",
    width=1200, height=1600, res=300)

pheatmap(
  mat,
  annotation_col = sample_anno,
  color = colorRampPalette(c("navy","white","firebrick3"))(100),
  cluster_rows = TRUE,
  cluster_cols = TRUE,
  fontsize_row = 7,
  fontsize_col = 9,
  main = "Top 30 DEGs: TreatedQ vs TreatedR"
)

dev.off()

message("✅ Heatmap generated successfully")
