#!/usr/bin/env Rscript

suppressPackageStartupMessages({
  library(DESeq2)
  library(pheatmap)
  library(AnnotationDbi)
  library(org.Mm.eg.db)   # mouse (use org.Hs.eg.db if human)
})

# -------- Arguments --------
args <- commandArgs(trailingOnly=TRUE)
results_dir <- ifelse(length(args) >= 1, args[1], "/mnt/c/RNASeqProject/results")
dds_file <- file.path(results_dir, "dds.rds")

if (!file.exists(dds_file)) {
  stop("❌ File not found: ", dds_file)
}

# -------- Load DESeq2 object --------
dds <- readRDS(dds_file)

# -------- Run DESeq2 (if not already run) --------
dds <- DESeq(dds)

# -------- Extract results --------
res <- results(dds)

# Remove NA
res <- res[!is.na(res$padj), ]

# -------- Clean Ensembl IDs --------
rownames(res) <- sub("\\..*", "", rownames(res))

# -------- Map gene symbols --------
gene_symbols <- mapIds(
  org.Mm.eg.db,
  keys = rownames(res),
  column = "SYMBOL",
  keytype = "ENSEMBL",
  multiVals = "first"
)

res$symbol <- gene_symbols

# Remove NA symbols
res <- res[!is.na(res$symbol), ]

# Remove duplicates (keep most significant)
res <- res[order(res$padj), ]
res <- res[!duplicated(res$symbol), ]

# -------- Filter significant DEGs --------
deg_sig <- res[res$padj < 0.05 & abs(res$log2FoldChange) > 1, ]

# -------- Select Top 30 DEGs --------
top30 <- head(deg_sig[order(deg_sig$padj), ], 30)
top30_genes <- rownames(top30)

# -------- VST --------
vst_mat <- assay(vst(dds, blind=TRUE))

# Clean Ensembl IDs in matrix
rownames(vst_mat) <- sub("\\..*", "", rownames(vst_mat))

# Subset matrix
mat <- vst_mat[top30_genes, ]

suppressPackageStartupMessages({
  library(org.Mm.eg.db)
  library(AnnotationDbi)
})

# Extract Ensembl IDs (remove version if present)
ensembl_ids <- rownames(mat)
ensembl_ids <- gsub("\\..*", "", ensembl_ids)

# Map to gene symbols
gene_symbols <- mapIds(
  org.Mm.eg.db,
  keys = ensembl_ids,
  column = "SYMBOL",
  keytype = "ENSEMBL",
  multiVals = "first"
)

# Replace rownames
rownames(mat) <- gene_symbols

# Remove NA symbols
mat <- mat[!is.na(rownames(mat)), ]

# Remove duplicates (keep first occurrence)
mat <- mat[!duplicated(rownames(mat)), ]

# Replace rownames with gene symbols
rownames(mat) <- top30$symbol

# -------- Scale per gene (Z-score) --------
mat <- t(scale(t(mat)))

# -------- Sample annotation --------
sample_anno <- as.data.frame(colData(dds)[, "group", drop=FALSE])
rownames(sample_anno) <- colnames(mat)

# -------- Heatmap colors --------
heat_colors <- colorRampPalette(c("navy","white","firebrick3"))(100)

# -------- Plot heatmap --------
pheatmap(
  mat,
  annotation_col = sample_anno,
  color = heat_colors,
  cluster_rows = TRUE,
  cluster_cols = TRUE,
  fontsize_row = 7,
  fontsize_col = 9,
  main = "Top 30 Differentially Expressed Genes"
)

# -------- Save heatmap --------
out_file <- file.path(results_dir, "heatmap_top30_DEGs.png")

png(out_file, width=1200, height=1600, res=300)

pheatmap(
  mat,
  annotation_col = sample_anno,
  color = heat_colors,
  cluster_rows = TRUE,
  cluster_cols = TRUE,
  fontsize_row = 7,
  fontsize_col = 9,
  main = "Top 30 Differentially Expressed Genes"
)

dev.off()

message("✅ Saved heatmap: ", out_file)
